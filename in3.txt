    public static void main(String[] a){
        System.out.println(new Aexp().Compute(-1,2,3));
        // System.out.println(new Aexp().Negate(true));
    }
}

class Aexp{
    public int Compute(int a, int b, int c){
        int num_aux;
        if (a < 0)
            num_aux = a * (b - 1) * c;
        else
            num_aux = a * c + b;
        return num_aux ;
    }

    public boolean Negate(boolean b) {
        System.out.println(b);
        return !b;
    }
}
